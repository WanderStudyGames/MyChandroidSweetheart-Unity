using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEditor;
using System;
[CustomEditor(typeof(Type))]

public class #SCRIPTNAME# : ExtendedEditor
{
    Texture _icon;
    string[] dependencies = new string[] { "" };
    //or
    string dependency = "";
    bool drawDependencies = false;
    private void OnEnable()
    {
        _icon = Resources.Load<Texture>("Ico_Scanner");
        drawDependencies = false;
        foreach (string s in dependencies)
        {
            if (serializedObject.FindProperty(s).objectReferenceValue == null) { drawDependencies = true; }
        }
        //or
        if (serializedObject.FindProperty(dependency).objectReferenceValue == null) { drawDependencies = true; }
    }
    public override void OnInspectorGUI()
    {
        serializedObject.Update();

        if (serializedObject.isEditingMultipleObjects)
        {
            base.OnInspectorGUI();
            return;
        }
        if (drawDependencies)
        {
            foreach (string s in dependencies)
            {
                DrawProperty(s);
            }
        }
        //or
        if (drawDependencies) { DrawProperty(dependency); }
        
        DrawAllProperties();

        serializedObject.ApplyModifiedProperties();
    }
}
